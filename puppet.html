Start

cd puppet.environments/

Create a new puppet branch

git checkout production
git pull
git checkout -b <branch name>
git push --set-upstream origin <branch name>

Commit changes to your branch

git add .
git commit -m "<message>"
git push

Test new changes on a server

sudo puppet agent --test --environment <branch> --noop

Merge your branch into Production

git checkout production
git pull
git checkout <branch>
git rebase -i production <branch>
(edit the file, using 'f' instead of 'pick' to squash unwanted commits)
git checkout production
git merge <branch> --no-ff
(write a meaningful merge message)
git push
git branch -d <branch>
git push origin --delete <branch>
